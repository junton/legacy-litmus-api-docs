Windows Mobile 6.0

 

 


<script type="text/javascript">
var imageTop = 0;

function scrollUp() {
        if (imageTop < 0)
        {
                imageTop += heightInc;
                $("screenimg").style.top = imageTop + "px";
        }
}

function scrollDown() {
        if((imageTop-100)>($("screenimg").height*-1) + heightInc)
        {
                imageTop -= heightInc;
                $("screenimg").style.top = imageTop + "px";
        }
}

function onKeyDown(e)
{
        var keycode;
        if (window.event) keycode = window.event.keyCode;
        else if (e) keycode = e.which;
        switch (keycode)
        {
                // UP
                case 38:
                                scrollUp();
                                return false;
                        break;
                // DOWN
                case 40:
                                scrollDown();
                                return false;
                        break;
        }
        return true;
}
document.onkeydown = onKeyDown;
</script>
